xs = []
ys1, ys2, ys3, ys4, ys5, ys6 = [], [],[], [],[], []


def animate(i):
    global s1, s2, s3
    xs.append(i)
    ys1.append(0.0)
    _, s1 = c1(torch.tensor([0.0]), s1)
    ys2.append(s1.v.detach().numpy())
    line_in1.set_data(xs, ys1)
    line_out1.set_data(xs, ys2)
    
    ys3.append(0.1)
    _, s2 = c2(torch.tensor([0.1]), s2)
    ys4.append(s2.v.detach().numpy())
    line_in2.set_data(xs, ys3)
    line_out2.set_data(xs, ys4)
    
    ys5.append(0.3)
    _, s3 = c3(torch.tensor([0.3]), s3)
    ys6.append(s3.v.detach().numpy())
    line_in3.set_data(xs, ys5)
    line_out3.set_data(xs, ys6)
    return line_in1, line_in2, line_in3

anim = animation.FuncAnimation(f, animate, interval=20, blit=True, frames=frames)
anim.save("images/spikes.gif", writer=animation.PillowWriter(fps=25))
